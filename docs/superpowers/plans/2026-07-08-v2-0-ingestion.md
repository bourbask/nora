# providers/trade_republic/scrape.py  (haut du fichier)
"""Trade Republic scrape wrapper (LOCAL, gitignoré).

Login interactif via pytr (PIN + 2FA), récupère la timeline, l'aplatit vers le
CSV ;-délimité que convert.py consomme déjà, écrit dans inbox/. Le timer
auto-sync fait ensuite convert + import (moitié non-interactive).

Secrets : TR_PHONE / TR_PIN depuis .env. pytr cache sa session (~/.pytr).
"""
CSV_COLUMNS = ["id", "timestamp", "title", "subtitle", "status",
               "eventType", "amount.value", "cashAccountNumber"]

def _amount_value(ev):
    amt = ev.get("amount")
    if not isinstance(amt, dict):
        return ""
    v = amt.get("value")
    return "" if v is None else str(v)

def flatten_events(events):
    """Raw pytr timeline events → list of row dicts keyed by CSV_COLUMNS.
    Missing fields → "". Amount object flattened to amount.value (dot-decimal)."""
    rows = []
    for ev in events:
        row = {c: str(ev.get(c, "") or "") for c in CSV_COLUMNS if c != "amount.value"}
        row["amount.value"] = _amount_value(ev)
        rows.append({c: row[c] for c in CSV_COLUMNS})   # ordre canonique
    return rows
```

    from pytr.account import login
    from pytr.timeline import Timeline

    tl = Timeline(tr)

    # Remplacer par la boucle de pagination confirmée au spike :
    async for ev in _iter_timeline(tr, tl):
        events.append(ev)
    return events

def write_inbox_csv(rows, inbox_dir):
    inbox = Path(inbox_dir); inbox.mkdir(parents=True, exist_ok=True)
    today = dt.date.today().isoformat()
    out = inbox / f"trade_republic_transactions_{today}.csv"
    with open(out, "w", newline="", encoding="utf-8") as f:
        w = csv.DictWriter(f, fieldnames=CSV_COLUMNS, delimiter=";")
        w.writeheader()
        w.writerows(rows)

# web/api/import_status.py
"""Dernier événement d'import par source, lu depuis data/auto_sync.log
(JSON-lines émis par scripts/auto-sync.sh). Lecture pure, hors chemin de l'argent."""
import json

def last_per_source(log_text):
    latest = {}
    for line in log_text.splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            ev = json.loads(line)
            src = ev["source"]
        except (json.JSONDecodeError, KeyError, TypeError):
            continue
        latest[src] = {"timestamp": ev.get("timestamp"),
                       "event": ev.get("event"),
                       "transactions": ev.get("transactions")}
    return latest
```

export function ago(iso: string | null): string {
  if (!iso) return "—";
  const s = Math.floor((Date.now() - new Date(iso).getTime()) / 1000);
  if (s < 3600) return `il y a ${Math.max(1, Math.floor(s / 60))} min`;
  if (s < 86400) return `il y a ${Math.floor(s / 3600)} h`;
  return `il y a ${Math.floor(s / 86400)} j`;
}
